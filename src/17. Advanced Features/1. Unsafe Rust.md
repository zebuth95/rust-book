# Unsafe Rust
Unsafe Rust allows you to perform operations that the compiler can't guarantee are safe. It's a way to opt out of Rust's safety checks when you know what you're doing.

### Four Unsafe Superpowers:

- Dereference raw pointers
- Call unsafe functions/methods
- Access or modify mutable static variables
- Implement unsafe traits


    // 1. RAW POINTERS
    fn raw_pointers_demo() {
        let mut num = 5;
    
        // Creating raw pointers (safe to create, unsafe to dereference)
        let r1 = &num as *const i32;
        let r2 = &mut num as *mut i32;
        
        // Can create raw pointers to arbitrary memory
        let address = 0x012345usize;
        let r3 = address as *const i32;
        
        unsafe {
            println!("r1 is: {}", *r1);
            println!("r2 is: {}", *r2);
            // println!("r3 is: {}", *r3); // Would segfault!
        }
    }
    
    // 2. UNSAFE FUNCTIONS
    unsafe fn dangerous() {
        println!("Doing something dangerous!");
    }
    
    unsafe trait DangerousTrait {
        fn dangerous_method(&self);
    }
    
    unsafe impl DangerousTrait for i32 {
        fn dangerous_method(&self) {
            println!("I'm a dangerous i32: {}", self);
        }
    }
    
    // Safe wrapper around unsafe code
    fn safe_abstraction() {
        let mut v = vec![1, 2, 3, 4, 5];
    
        let r = &mut v[..];
        let (a, b) = r.split_at_mut(3);
        
        println!("First part: {:?}", a);
        println!("Second part: {:?}", b);
    }
    
    // Manual implementation of split_at_mut
    fn split_at_mut(slice: &mut [i32], mid: usize) -> (&mut [i32], &mut [i32]) {
        let len = slice.len();
        let ptr = slice.as_mut_ptr();
    
        assert!(mid <= len);
        
        unsafe {
            (
                std::slice::from_raw_parts_mut(ptr, mid),
                std::slice::from_raw_parts_mut(ptr.add(mid), len - mid),
            )
        }
    }
    
    // 3. EXTERNAL FUNCTIONS (FFI)
    extern "C" {
        fn abs(input: i32) -> i32;
        }
    
    fn call_c_function() {
        unsafe {
            println!("Absolute value of -3 according to C: {}", abs(-3));
        }
    }
    
    // 4. MUTABLE STATIC VARIABLES
    static mut COUNTER: u32 = 0;
    
    fn add_to_count(inc: u32) {
        unsafe {
            COUNTER += inc;
        }
    }
    
    fn static_mutable_demo() {
        add_to_count(3);
    
        unsafe {
            println!("COUNTER: {}", COUNTER);
        }
    }
    
    fn main() {
        raw_pointers_demo();
    
        unsafe {
            dangerous();
        }
        
        42.dangerous_method();
        
        safe_abstraction();
        call_c_function();
        static_mutable_demo();
    }
