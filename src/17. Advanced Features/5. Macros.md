# Macros
Macros are a way of writing code that writes other code (metaprogramming). Rust has:

- Declarative Macros: Pattern-based code generation
- Procedural Macros: Custom derives, attribute-like, function-like

### Basic Declarative Macros

    // Simple macro that greets
    macro_rules! say_hello {
        () => {
            println!("Hello!");
        };
    }
    
    // Macro with parameters
    macro_rules! greet {
        ($name:expr) => {
            println!("Hello, {}!", $name);
        };
    }
    
    // Macro that works like a function
    macro_rules! add {
        ($a:expr, $b:expr) => {
            $a + $b
        };
    }
    
    fn basic_macros() {
        say_hello!();
        greet!("Alice");
        greet!("Bob");
        
        let result = add!(5, 3);
        println!("5 + 3 = {}", result);
    }

### Macros with Repetition:

    // Macro that creates a vector
    macro_rules! my_vec {
        ($($element:expr),*) => {
        {
            let mut temp_vec = Vec::new();
            $(
            temp_vec.push($element);
            )*
            temp_vec
            }
        };
    }
    
    // Macro that prints multiple values
    macro_rules! print_all {
        ($($value:expr),*) => {
            $(
                println!("{}", $value);
            )*
        };
    }
    
    fn repetition_macros() {
        let numbers = my_vec![1, 2, 3, 4, 5];
        println!("Created vector: {:?}", numbers);
    
        print_all!("apple", "banana", "cherry");
    }

### Conditional Compilation Macro

    // Macro that only runs in debug mode
    macro_rules! debug_log {
        ($message:expr) => {
            if cfg!(debug_assertions) {
                println!("[DEBUG] {}", $message);
            }
        };
    }
    
    // Macro for different environments
    macro_rules! environment_specific {
        ("dev") => {
            println!("Running in development mode");
        };
        ("prod") => {
            println!("Running in production mode");
        };
        ($other:expr) => {
            println!("Unknown environment: {}", $other);
        };
    }
    
    fn conditional_macros() {
        debug_log!("This is a debug message");
    
        environment_specific!("dev");
        environment_specific!("prod");
        environment_specific!("test");
    }

### HTML Builder Macro (Practical Example)

    // Simple HTML tag macro
    macro_rules! html_tag {
        ($tag:ident, $content:expr) => {
            format!("<{}>{}</{}>", stringify!($tag), $content, stringify!($tag))
        };
    
        ($tag:ident) => {
            format!("<{}/>", stringify!($tag))
        };
    }
    
    fn html_macro() {
        let paragraph = html_tag!(p, "This is a paragraph");
        let line_break = html_tag!(br);
        
        println!("{}", paragraph);
        println!("{}", line_break);
    }

### Assertion Macros

    // Custom assert macro with better messages
    macro_rules! my_assert {
        ($condition:expr) => {
            if !$condition {
                panic!("Assertion failed: {}", stringify!($condition));
            }
        };
    
        ($condition:expr, $message:expr) => {
            if !$condition {
                panic!("{}", $message);
            }
        };
    }
    
    fn assertion_macros() {
        let x = 5;
        let y = 10;
    
        my_assert!(x < y);
        my_assert!(x == 5, "x should be 5");
        
        // This would panic:
        // my_assert!(x > y, "x should be greater than y");
    }

### Simple Calculator Macro

    // Macro that acts like a simple calculator
    macro_rules! calculate {
        // Addition
        ($a:expr + $b:expr) => {
            $a + $b
        };
    
        // Subtraction  
        ($a:expr - $b:expr) => {
            $a - $b
        };
        
        // Multiplication
        ($a:expr * $b:expr) => {
            $a * $b
        };
        
        // Division
        ($a:expr / $b:expr) => {
            $a / $b
        };
    }
    
    fn calculator_macro() {
        let sum = calculate!(10 + 5);
        let difference = calculate!(20 - 8);
        let product = calculate!(6 * 7);
        let quotient = calculate!(15 / 3);
    
        println!("10 + 5 = {}", sum);
        println!("20 - 8 = {}", difference);
        println!("6 * 7 = {}", product);
        println!("15 / 3 = {}", quotient);
    }

### Database Query Macro (Conceptual)

    // Macro that simulates database operations
    macro_rules! db_query {
        (SELECT * FROM $table:ident WHERE $field:ident = $value:expr) => {
        format!("SELECT * FROM {} WHERE {} = {}", stringify!($table), stringify!($field), $value)
        };
    
        (INSERT INTO $table:ident VALUES $($val:expr),*) => {
            {
                let mut query = format!("INSERT INTO {} VALUES ", stringify!($table));
                $(
                    query.push_str(&format!("{}, ", $val));
                )*
                query.pop(); // Remove last comma
                query.pop(); // Remove last space
                query
            }
        };
    }
    
    fn db_macro() {
        let select_query = db_query!(SELECT * FROM users WHERE id = 123);
        let insert_query = db_query!(INSERT INTO users VALUES "Alice", 25, "Engineer");
    
        println!("{}", select_query);
        println!("{}", insert_query);
    }
