# Advanced Functions and Closures
Advanced function and closure features include function pointers, returning closures, and advanced closure traits.

    // 1. FUNCTION POINTERS
    fn add_one(x: i32) -> i32 {
        x + 1
    }
    
    fn do_twice(f: fn(i32) -> i32, arg: i32) -> i32 {
        f(f(arg))
    }
    
    // Function pointers vs closures
    fn function_pointer_demo() {
        let answer = do_twice(add_one, 5);
        println!("The answer is: {}", answer);
    
        // Closures that don't capture environment can be used as function pointers
        let closure = |x: i32| x + 1;
        let answer2 = do_twice(closure, 5);
        println!("The answer is still: {}", answer2);
    }
    
    // 2. RETURNING CLOSURES
    // Method 1: Using Box
    fn returns_closure_box() -> Box<dyn Fn(i32) -> i32> {
        Box::new(|x| x + 1)
    }
    
    // Method 2: Using impl Trait (preferred)
    fn returns_closure_impl() -> impl Fn(i32) -> i32 {
        |x| x + 1
    }
    
    // Method 3: For closures that capture environment
    fn make_adder(y: i32) -> impl Fn(i32) -> i32 {
        move |x| x + y
    }
    
    // 3. CLOSURE TRAITS: Fn, FnMut, FnOnce
    struct Cache<T>
    where
        T: Fn(u32) -> u32,
    {
        calculation: T,
        value: Option<u32>,
    }
    
    impl<T> Cache<T>
    where
        T: Fn(u32) -> u32,
    {
        fn new(calculation: T) -> Cache<T> {
            Cache {
                calculation,
                value: None,
            }
        }
    
        fn value(&mut self, arg: u32) -> u32 {
            match self.value {
                Some(v) => v,
                None => {
                    let v = (self.calculation)(arg);
                    self.value = Some(v);
                    v
                }
            }
        }
    }
    
    // 4. ADVANCED CLOSURE USAGE
    fn advanced_closures_demo() {
        // Closures with different trait bounds
        let immutable_closure = |x: i32| println!("Got: {}", x);
        let mut mutable_closure = |x: i32| {
        let mut counter = 0;
        counter += x;
        counter
        };
        let once_closure = |x: String| x;
    
        // Using the closures
        immutable_closure(5);
        let result = mutable_closure(3);
        println!("Mutable closure result: {}", result);
        let string_result = once_closure(String::from("hello"));
        println!("Once closure result: {}", string_result);
        
        // Cache example
        let expensive_closure = |num: u32| -> u32 {
            println!("Calculating slowly...");
            std::thread::sleep(std::time::Duration::from_secs(1));
            num
        };
        
        let mut cached_result = Cache::new(expensive_closure);
        println!("First call: {}", cached_result.value(10));
        println!("Cached call: {}", cached_result.value(10)); // Uses cached value
    }

    // 5. CLOSURES WITH COMPLEX CAPTURES
    fn closure_captures_demo() {
        let x = 4;
        let equal_to_x = |z| z == x;  // Captures x immutably
    
        let y = 4;
        assert!(equal_to_x(y));
        
        // Moving ownership into closure
        let data = vec![1, 2, 3];
        let closure = move || {
            println!("Data: {:?}", data);
            // data is moved here, can't be used after
        };
        closure();
        // println!("{:?}", data); // This would error - data was moved
        
        // FnMut example
        let mut list = vec![1, 2, 3];
        println!("Before: {:?}", list);
        
        let mut borrows_mutably = || list.push(7);
        borrows_mutably();
        println!("After: {:?}", list);
    }

    fn main() {
        function_pointer_demo();
        advanced_closures_demo();
        closure_captures_demo();
    
        // Returning closures
        let adder = make_adder(5);
        println!("5 + 10 = {}", adder(10));
        
        let closure_box = returns_closure_box();
        println!("Boxed closure: {}", closure_box(5));
        
        let closure_impl = returns_closure_impl();
        println!("Impl closure: {}", closure_impl(5));
    }
